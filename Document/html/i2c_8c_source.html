690 {

692 
693  I2C_START(i2c);
694  while(u8Xfering && (u8Err == 0u))
695  {

701  u8Ctrl = I2C_CTL_SI; /* Clear SI */
702  break;

717  I2C_SET_CONTROL_REG(i2c, I2C_CTL_STO_SI); /* Clear SI and send STOP */
